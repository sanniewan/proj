# Use a Python base image (choose arm64 if needed)
FROM python:3.10-slim

# Install system dependencies (if needed for board, e.g., I2C/SPI)
RUN apt-get update && \
    apt-get install -y python3-dev libgpiod2 && \
    rm -rf /var/lib/apt/lists/*

# Install board and related libraries
RUN pip install --no-cache-dir board

# Copy your application code
WORKDIR /app
COPY ./src .

# Set default command
CMD ["python3", "read_sensor.py"]
